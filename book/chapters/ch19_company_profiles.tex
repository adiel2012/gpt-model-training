\chapter{How Top Companies Train Their Models}

\section{OpenAI -- GPT-5 Family}

\begin{companybox}[OpenAI -- GPT-5 / o3 / o4-mini]
\textbf{Architecture:} Dense transformer (rumored MoE), Azure AI supercomputers. Unified intelligence, reasoning, coding, multimodality.

\textbf{Pre-training:} Massive-scale unsupervised learning. GPT-4.5: data from smaller models. GPT-5: 94.6\% AIME, 74.9\% SWE-bench, 84.2\% MMMU.

\textbf{Post-training:} SFT + RLHF + ``safe completions'' (output-centric safety). Reasoning via dedicated RL chains of thought. Process reward models score intermediate reasoning steps.

\textbf{Deployment:} Real-time router (instant vs.\ deep reasoning). $\sim$80\% fewer factual errors, 50--80\% fewer output tokens.
\end{companybox}

\section{DeepSeek -- R1 and V3}

\begin{companybox}[DeepSeek -- R1 / V3]
\textbf{Architecture:} Fine-grained MoE, shared expert isolation, multi-head latent attention (MLA).

\textbf{Pre-training:} Trillions of tokens at $\sim$\$5.5M.

\textbf{R1 pipeline \parencite{guo2025deepseekr1} (4 stages):}
\begin{enumerate}[leftmargin=*]
  \item Cold-Start SFT (thousands of examples)
  \item RLVR + GRPO (16 responses/prompt, rule-based rewards, no neural RM)
  \item Rejection Sampling + SFT Stage 2 (800K filtered examples)
  \item Final RL: RLVR (reasoning) + RLHF (helpfulness/safety with separate RMs)
\end{enumerate}

\textbf{Results:} R1-Zero AIME: 15.6\% $\rightarrow$ 71.0\% via pure RL. Emergent self-verification, ``aha moments.''

\textbf{Distillation:} 1.5B--70B models. 7B: 55.5\% AIME 2024.
\end{companybox}

\section{Meta -- Llama 4 Family}

\begin{companybox}[Meta -- Llama 4 Scout / Maverick / Behemoth]
\textbf{Architecture:} First Llama MoE. Scout: 17B/109B (16 exp.). Maverick: 17B/400B (128 exp.). Behemoth: 288B/$\sim$2T. iRoPE: 10M-token context.

\textbf{Pre-training:} Early fusion multimodality (text + image + video from start). MetaCLIP vision encoder.

\textbf{Post-training (key insight: heavy SFT hurts RL):}
\begin{itemize}[leftmargin=*]
  \item Lightweight SFT: removed 50\%+ easy examples (Llama-as-judge)
  \item Online RL: continuous, adaptive filtering, curriculum design
  \item Lightweight DPO: final polish
  \item Behemoth codistillation: teacher $\rightarrow$ Scout/Maverick
\end{itemize}

\textbf{Deployment:} Scout on 1$\times$H100 (INT4). 9--23$\times$ price-performance vs.\ GPT-4o.
\end{companybox}

\section{Anthropic -- Claude Family}

\begin{companybox}[Anthropic -- Claude 4 Family]
\textbf{Architecture:} Dense transformer. Details not publicly disclosed.

\textbf{Constitutional AI (CAI):}
\begin{itemize}[leftmargin=*]
  \item Phase 1 (SL-CAI): Generate $\rightarrow$ self-critique against principles $\rightarrow$ revise. Revised outputs become SFT data.
  \item Phase 2 (RL-CAI): Revised outputs train preference model for RLHF. RLAIF reduces human annotation needs.
\end{itemize}

\textbf{Safety:} Classifiers, harmlessness RMs, extensive red-teaming. Honest, harmless, helpful.
\end{companybox}

\section{Google DeepMind -- Gemini Family}

\begin{companybox}[Google DeepMind -- Gemini 2.5]
\textbf{Architecture:} Natively multimodal transformer on TPU v5. Text, images, audio, video, code unified.

\textbf{Post-training:} SFT + RLHF + reasoning RL + ``thinking'' mode. Distillation: Pro $\rightarrow$ Flash $\rightarrow$ Nano.

\textbf{Deployment:} Google products, Vertex AI, on-device Nano.
\end{companybox}

\section{Alibaba / Qwen -- Qwen 3 Family}

\begin{companybox}[Alibaba -- Qwen 3 / Qwen 3-VL]
\textbf{Architecture:} Dense + MoE. Qwen3-VL: vision-language MoE (30B-A3B, 235B-A22B).

\textbf{Post-training:} GSPO (Group Sequence Policy Optimization). ``Thinking'' toggle.

\textbf{Open ecosystem:} Full open weights 0.6B--235B. Popular base for R1 distillation.
\end{companybox}

\section{Cross-Cutting Patterns}

\begin{patternbox}[Converging Trends Across Frontier Labs]
\begin{itemize}
  \item \textbf{Modular post-training:} Distinct stages, not monolithic. Heavy SFT hurts RL (Meta).
  \item \textbf{MoE is default:} Meta, DeepSeek, Qwen, rumored closed labs.
  \item \textbf{GRPO displaces PPO:} Critic-free, 33--50\% memory savings.
  \item \textbf{Distillation is first-class:} Behemoth$\rightarrow$Scout, R1$\rightarrow$1.5B--70B, Pro$\rightarrow$Flash$\rightarrow$Nano.
  \item \textbf{Reasoning via RL:} Pure RL (DeepSeek), unified routing (OpenAI), thinking modes (Google/Qwen).
  \item \textbf{Safety diverges productively:} Safe completions, CAI, Llama-as-judge, separate RMs.
\end{itemize}
\end{patternbox}
